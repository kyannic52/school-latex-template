\begin{infobox}[Ziel der Stunde]
Du erkennst lineare Funktionen der Form $y=mx+b$ und kannst ihre Graphen zeichnen.
\end{infobox}

\begin{definitionbox}[Merke]
Bei $y=mx+b$ beschreibt $m$ die Steigung und $b$ den Schnittpunkt mit der $y$-Achse.
\end{definitionbox}

\begin{task}[difficulty=1,type=xox,points=4,tag=grundlagen,title=Warm-up]
Bestimme jeweils Steigung $m$ und Achsenabschnitt $b$.
\begin{enumerate}
\item $y=2x+3$
\item $y=-\frac12x+4$
\item $y=7x-1$
\end{enumerate}
\end{task}

\begin{task}[difficulty=2,type=xxo,points=6,tag=graph,title=Graph zeichnen]
Zeichne den Graphen der Funktion $f(x)=\frac12x-2$ in ein Koordinatensystem.
\answerlines[4]
\end{task}

\begin{task}[difficulty=3,type=oox,points=8,tag=modellieren,title=Knobelaufgabe]
Eine Gerade geht durch die Punkte $A(2|1)$ und $B(6|9)$. Bestimme die Funktionsgleichung.
\end{task}
